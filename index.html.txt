<script>
(function(){

const adminUser = "admin";
const adminPass = "1337";

let xp = parseInt(localStorage.getItem("xp")) || 0;
let cheats = JSON.parse(localStorage.getItem("cheats")) || [];
let loggedIn = localStorage.getItem("loggedIn") === "true";
let shutdown = localStorage.getItem("shutdown") === "true";

const loginBox = document.getElementById("loginBox");
const adminPanel = document.getElementById("adminPanel");
const shutdownScreen = document.getElementById("shutdownScreen");
const xpDisplay = document.getElementById("xp");
const cheatsDiv = document.getElementById("cheats");

function save(){
    localStorage.setItem("xp", xp);
    localStorage.setItem("cheats", JSON.stringify(cheats));
    localStorage.setItem("loggedIn", loggedIn);
    localStorage.setItem("shutdown", shutdown);
}

function render(){
    xpDisplay.textContent = xp;
    cheatsDiv.innerHTML = "";
    cheats.forEach(c=>{
        const div = document.createElement("div");
        div.className = "cheat-card";
        div.textContent = c;
        cheatsDiv.appendChild(div);
    });

    loginBox.classList.toggle("hidden", loggedIn);
    adminPanel.classList.toggle("hidden", !loggedIn);
    shutdownScreen.classList.toggle("hidden", !shutdown);
}

document.getElementById("loginBtn").onclick = function(){
    const u = document.getElementById("username").value;
    const p = document.getElementById("password").value;

    if(u === adminUser && p === adminPass){
        loggedIn = true;
        save();
        render();
    } else {
        document.getElementById("message").textContent = "Wrong login";
    }
};

document.getElementById("logoutBtn").onclick = function(){
    loggedIn = false;
    save();
    render();
};

document.getElementById("addCheatBtn").onclick = function(){
    const cheat = prompt("Enter cheat:");
    if(cheat){
        cheats.push(cheat);
        xp += 10;
        save();
        render();
    }
};

document.getElementById("shutdownBtn").onclick = function(){
    shutdown = true;
    save();
    render();
};

document.getElementById("restoreBtn").onclick = function(){
    shutdown = false;
    save();
    render();
};

render();

})();
</script>
